// Riverpod 2.x — generator syntax, zero boilerplate
@riverpod
Stream<UserEntity?> authState(AuthStateRef ref) {
  return ref.watch(authRepositoryProvider).authStateChanges;
}

@riverpod
class AuthNotifier extends _$AuthNotifier {
  @override
  AsyncValue<void> build() => const AsyncData(null);

  Future<Either<Failure, UserEntity>> signIn({
    required String email,
    required String password,
  }) async {
    state = const AsyncLoading();

    final result = await SignInWithEmailUseCase(_repository)
        .call(SignInParams(email: email, password: password));

    result.fold(
      (failure) => state = AsyncError(failure, StackTrace.current),
      (_)       => state = const AsyncData(null),
    );
    return result;
  }

  Future<void> signOut() async {
    state = const AsyncLoading();
    await _repository.signOut();
    state = const AsyncData(null);
  }
}

// Firebase errors never escape to the UI
Failure _mapFirebaseAuthException(FirebaseAuthException e) {
  return switch (e.code) {
    'user-not-found'      => const InvalidCredentialsFailure(),
    'email-already-in-use' => const EmailAlreadyInUseFailure(),
    'weak-password'       => const WeakPasswordFailure(),
    'network-request-failed' => const NetworkFailure(),
    _ => AuthFailure(message: e.message ?? 'Auth failed'),
  };
}
// GoRouter — reactive auth guard + indexed shell
@riverpod
GoRouter appRouter(AppRouterRef ref) {
  final authState = ref.watch(authStateProvider);

  return GoRouter(
    redirect: (context, state) {
      final isAuth = authState.valueOrNull != null;
      final isAuthRoute = state.matchedLocation.startsWith('/auth');

      if (!isAuth && !isAuthRoute) return AppRoutes.login;
      if (isAuth && isAuthRoute)  return AppRoutes.home;
      return null;
    },
    routes: [
      StatefulShellRoute.indexedStack(
        builder: (ctx, state, shell) =>
            ScaffoldWithNavBar(shell),
        branches: [
          StatefulShellBranch(routes: [homeRoute]),
          StatefulShellBranch(routes: [profileRoute]),
          StatefulShellBranch(routes: [settingsRoute]),
        ],
      ),
    ],
  );
}
// Typed failures — no String errors, ever
abstract class Failure extends Equatable {
  const Failure({ required this.message, this.code });
  final String message;
  final String? code;
}

Features

Everything you need.
Nothing you don't.

Every decision stress-tested in real production apps with real users and real consequences.

⚡
Riverpod 2.x + Code Gen

Generator-based providers with @riverpod annotations. AsyncNotifier for actions, StreamProvider for real-time. Zero boilerplate, compile-time safety.

🧭
GoRouter with Auth Guards

Declarative navigation with reactive redirect guards, type-safe routes via AppRoutes, deep linking, and StatefulShellRoute for bottom navigation.

🔥
Full Firebase Suite

Auth (email, Google, Apple), Firestore, Storage, Analytics, Crashlytics, Remote Config, and Push Notifications — 8 services pre-wired and working.

🏗️
Feature-First Architecture

Domain entities are pure Dart. Data models know Firebase. Presentation only touches providers. Strict layer contracts — impossible to violate by accident.

🛡️
Typed Error Handling

Either<Failure, T> from dartz on every repository call. 15 typed Failure classes. No raw exceptions leaking from Firebase into your UI.

🌐
Dio + 4 Interceptors

Auth injection, automatic 401 token refresh, exponential backoff retry, typed error mapping, and full request/response logging — all pre-configured.

🎨
Material 3 Theme System

Complete light/dark theme, Inter font, full color token system, every component styled consistently, and persistent theme preference via SharedPreferences.

🧪
Tests Included

6 test files covering use cases, entities, failures, and validators. Structure mirrors the feature folders. mocktail pre-configured for mocking repositories.

🔧
Makefile + Linting

One-command setup, codegen, formatting, coverage. very_good_analysis + riverpod_lint + custom_lint catch problems before they reach production.

01

Data flows inward only

Presentation never imports from data. Data never imports from presentation. Domain has zero external dependencies — pure Dart classes, always.

02

One use case per operation

Each use case does exactly one thing. SignInWithEmailUseCase, GetPostsUseCase, ToggleLikeUseCase. Small, focused, easy to test in isolation.

03

Repositories as contracts

The domain defines what the repository must do. The data layer defines how. Swap Firebase for Supabase without touching a single use case.

04

New feature = same folder pattern

Every developer on your team adds features the same way. The structure enforces consistency. No debates about where something lives.

Quick Start

Zero to running
in under 5 minutes.

1

Clone the template

Hit "Use This Template" on GitHub, name your repo, and clone it locally. Or git clone directly.

2

Install + generate code

Run make setup — this runs flutter pub get and build_runner in one shot. All generated files will appear.

3

Configure Firebase

Run flutterfire configure with the FlutterFire CLI. It auto-generates firebase_options.dart with your project's real keys.

4

Set your environment

Copy .env.example.json → .env.json, set your BASE_URL, then run with make run. Done.

bash 
# 1. Clone
$ git clone github.com/eng-kerolos-adel/flutter-riverpod-template my_app
$ cd my_app

# 2. Install + codegen
$ make setup
✓ flutter pub get
✓ build_runner complete

# 3. Connect Firebase
$ flutterfire configure
✓ firebase_options.dart generated

# 4. Configure env
$ cp .env.example.json .env.json

# 5. Run!
$ make run
✓ Launching on Chrome...
✓ Running on Android...
